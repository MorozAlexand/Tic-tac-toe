board = list(range(1, 10))

# создал функцию в которой
# выводим первую строку состоящую из 13 символов «тире», после чего, в цикле прорисовываем остальные края поля.
def draw_board(board):
    print("-" * 13)
    for i in range(3):
        print("|", board[0 + i * 3], "|", board[1 + i * 3], "|", board[2 + i * 3], "|")
        print("-" * 13)
